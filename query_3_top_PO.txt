

WITH MonthlyCounts AS (
    -- Calculate the number of POs for every month in the target years
    SELECT 
        EXTRACT(YEAR FROM Order_date) AS Order_Year,
        EXTRACT(MONTH FROM Order_date) AS Order_Month,
        COUNT(OID) AS PO_Count
    FROM PO
    WHERE Order_date >= '2024-01-01' AND Order_date <= '2025-12-31'
    GROUP BY 1, 2
),
RankedMonths AS (
    -- Rank the months within each individual year
    SELECT 
        Order_Year, 
        Order_Month, 
        PO_Count,
        ROW_NUMBER() OVER(PARTITION BY Order_Year ORDER BY PO_Count DESC) as monthly_rank
    FROM MonthlyCounts
)
-- Filter for the top 3 of each year
SELECT 
    Order_Year, 
    Order_Month, 
    PO_Count
FROM RankedMonths
WHERE monthly_rank <= 3
ORDER BY Order_Year DESC, PO_Count DESC;
